```typescript
import { defineConfig } from "vite";
import react from "@vitejs/plugin-react-swc";
import tailwindcss from "@tailwindcss/vite";
import { lingui } from "@lingui/vite-plugin";
import path from "path";

  plugins: [
    react({
      plugins: [["@lingui/swc-plugin", {}]],

    tailwindcss(),
    lingui(),

});
```

```css
    background: repeating-linear-gradient(

  }

```

Important: Use `@theme` (NOT `@theme inline`). Per RESEARCH.md Pitfall 2, `@theme inline` is for runtime CSS variable indirection -- retro tokens are static hex values.

Do NOT create a `tailwind.config.js` or `tailwind.config.ts` file. Tailwind v4 uses CSS-first configuration.
