    **A.3. Create `frontend2/e2e/scan-lookup.spec.ts`** (THE regression test for G-65-01)

    ```typescript
    // Phase 65 Plan 65-11 — G-65-01 regression test.
    //
    // Why this test exists: all 710 Vitest unit tests passed with G-65-01
    // present in production because they mocked above the backend boundary.
    // Firefox MCP UAT on 2026-04-19 found EI LEITUD when MATCHED was expected,
    // traced to backend FTS search_vector missing barcode coverage. The fix
    // (Plans 65-09 + 65-10) adds a dedicated /items/by-barcode/{code} endpoint
    // and swaps the frontend to call it. THIS test catches any revert.
    //
    // Contract: runs against a real backend + real Postgres. Seeder user is
    // assumed to exist (seeder@test.local); adjust if the project uses
    // different dev credentials.
    import { test, expect } from "@playwright/test";

    const TEST_USER = process.env.E2E_USER ?? "seeder@test.local";
    const TEST_PASS = process.env.E2E_PASS ?? "password";
    const TEST_BARCODE = `E2E-${Date.now()}`;  // unique per run, avoids collisions
    const TEST_NAME = `G-65-01 Regression Item ${Date.now()}`;

    test("G-65-01 regression: scanned barcode resolves to workspace item via MATCH banner", async ({ page, request }) => {
      // 1. Log in via the UI so the session cookie is set for both page + request contexts.
      await page.goto("/auth/login");
      await page.getByLabel(/email/i).fill(TEST_USER);
      await page.getByLabel(/password/i).fill(TEST_PASS);
      await page.getByRole("button", { name: /log in|sign in/i }).click();

      // 2. Wait for any post-login route (anything outside /auth/*). This is
      //    future-proof against new post-login landing routes (dashboard,
      //    items, scan, /, or something new) — we just want "not /auth".
      await page.waitForURL((url) => !url.pathname.startsWith("/auth"), {
        timeout: 10_000,
      });

      // 3. Create an item with the known barcode via the same cookie-authenticated
      //    request context the browser holds.
      const wsId = await page.evaluate(() => {
        // Pull workspaceId from whatever global exposes it — check AuthContext or localStorage.
        // If none, fall back to parsing from the URL. This block is the most
        // project-specific piece; adjust after inspecting AuthContext runtime shape.
        const match = window.location.pathname.match(/\/workspaces\/([^/]+)/);
        return match?.[1] ?? localStorage.getItem("hws-workspace-id") ?? "";
      });
      expect(wsId, "workspace id must be resolvable for seeding").not.toBe("");

      const createResp = await request.post(`/api/workspaces/${wsId}/items`, {
        data: {
          name: TEST_NAME,
          sku: `E2E-SKU-${Date.now()}`,
          barcode: TEST_BARCODE,
          min_stock_level: 0,
        },
      });
      expect(createResp.status(), "item seed must succeed").toBe(200);
      const created = (await createResp.json()) as { id: string; barcode: string };
      expect(created.barcode).toBe(TEST_BARCODE);

      try {
        // 4. Navigate to /scan and open the manual-entry tab.
        await page.goto("/scan");
        await page.getByRole("tab", { name: /käsitsi|manual/i }).click();

        // 5. Enter the barcode and submit.
        await page.getByRole("textbox").fill(TEST_BARCODE);
        await page.getByRole("button", { name: /otsi kood|lookup|search/i }).click();

        // 6. ASSERTION: MATCHED banner appears with the item name.
        // Both EN and ET labels accepted to avoid locale flakiness.
        await expect(
          page.getByText(/matched|vaste leitud/i),
        ).toBeVisible({ timeout: 5_000 });
        await expect(page.getByText(TEST_NAME)).toBeVisible();

        // 7. Sanity: the VIEW ITEM button navigates to /items/{id}.
        await page.getByRole("button", { name: /view item|vaata eset/i }).click();
        await page.waitForURL(new RegExp(`/items/${created.id}`));
      } finally {
        // 8. Cleanup — best-effort.
        await request.delete(`/api/workspaces/${wsId}/items/${created.id}`);
      }
    });
    ```

    **B.2. Create `backend/internal/domain/warehouse/item/handler_integration_test.go`**

    ```go
    //go:build integration
    // +build integration

    package item_test

    import (
        "context"
        "encoding/json"
        "fmt"
        "net/http"
        "os"
        "testing"

        "github.com/google/uuid"
        "github.com/stretchr/testify/require"

        "github.com/antti/home-warehouse/go-backend/internal/domain/warehouse/item"
        "github.com/antti/home-warehouse/go-backend/internal/infra/postgres"
        "github.com/antti/home-warehouse/go-backend/internal/testutil"
    )

    // G-65-01 regression: exercises the real repo + real Postgres behind the
    // /items/by-barcode/{code} route. Catches backend-only regressions of
    // Plan 65-09 (endpoint reverted, FindByBarcode wired wrong, etc.).
    //
    // Frontend-side regressions (Plan 65-10 revert) are NOT caught here —
    // that was the tradeoff noted in Plan 65-11 Task 0 Option B cons.
    func TestItemHandler_LookupByBarcode_Integration(t *testing.T) {
        dbURL := os.Getenv("TEST_DATABASE_URL")
        if dbURL == "" {
            t.Skip("TEST_DATABASE_URL not set; skipping integration test")
        }

        pool := testutil.MustOpenTestDB(t, dbURL)
        defer pool.Close()

        // Use existing NewHandlerTestSetup + swap the mock service for a real
        // service backed by the real postgres repo. If NewHandlerTestSetup
        // does not support a real service, build the chi router + huma API
        // inline here with the same middleware.
        ctx := context.Background()
        workspaceID := testutil.SeedWorkspace(t, pool)
        defer testutil.TruncateItems(t, pool, workspaceID)

        repo := postgres.NewItemRepository(pool)
        svc := item.NewService(repo, /* category repo stub */ nil)

        setup := testutil.NewIntegrationHandlerSetup(t, workspaceID)
        item.RegisterRoutes(setup.API, svc, nil, nil, nil)

        barcode := fmt.Sprintf("INT-TEST-%s", uuid.New().String()[:8])
        seededItem := testutil.InsertItem(t, pool, workspaceID, item.CreateInput{
            WorkspaceID: workspaceID,
            Name:        "Integration Test Cola",
            SKU:         "INT-SKU-" + uuid.New().String()[:8],
            Barcode:     &barcode,
        })

        t.Run("returns 200 with the seeded item on exact barcode match", func(t *testing.T) {
            rec := setup.Get(fmt.Sprintf("/items/by-barcode/%s", barcode))
            require.Equal(t, http.StatusOK, rec.Code)

            var resp item.ItemResponse
            require.NoError(t, json.Unmarshal(rec.Body.Bytes(), &resp))
            require.Equal(t, seededItem.ID(), resp.ID)
            require.Equal(t, workspaceID, resp.WorkspaceID)
            require.NotNil(t, resp.Barcode)
            require.Equal(t, barcode, *resp.Barcode)
        })

        t.Run("returns 404 for non-existent barcode", func(t *testing.T) {
            rec := setup.Get("/items/by-barcode/NEVER-EXISTED-" + uuid.New().String()[:8])
            require.Equal(t, http.StatusNotFound, rec.Code)
        })

        t.Run("returns 404 for barcode in another workspace (Pitfall #5 cross-tenant scoping)", func(t *testing.T) {
            otherWorkspaceID := testutil.SeedWorkspace(t, pool)
            defer testutil.TruncateItems(t, pool, otherWorkspaceID)

            otherBarcode := fmt.Sprintf("OTHER-WS-%s", uuid.New().String()[:8])
            testutil.InsertItem(t, pool, otherWorkspaceID, item.CreateInput{
                WorkspaceID: otherWorkspaceID,
                Name:        "Other Workspace Item",
                SKU:         "OTHER-SKU-" + uuid.New().String()[:8],
                Barcode:     &otherBarcode,
            })

            // Query from OUR workspace (setup.WorkspaceID === workspaceID) for
            // the barcode that only exists in otherWorkspaceID — must be 404.
            rec := setup.Get(fmt.Sprintf("/items/by-barcode/%s", otherBarcode))
            require.Equal(t, http.StatusNotFound, rec.Code,
                "cross-tenant barcode leak — FindByBarcode WHERE clause must scope by workspace_id")
        })
    }
    ```

    Note: if `testutil.MustOpenTestDB`, `testutil.SeedWorkspace`, `testutil.TruncateItems`, `testutil.InsertItem`, or `testutil.NewIntegrationHandlerSetup` do NOT exist, create them in `backend/internal/testutil/integration.go` (also behind the `//go:build integration` tag).
