       import { forwardRef, useCallback, type TextareaHTMLAttributes, type ChangeEvent } from "react";

         error?: string;
       }

       const MAX_ROWS = 8;
       const LINE_HEIGHT = 24; // 16px * 1.5

       const RetroTextarea = forwardRef<HTMLTextAreaElement, RetroTextareaProps>(
         ({ error, className, onInput, ...rest }, ref) => {
           const handleInput = useCallback((e: ChangeEvent<HTMLTextAreaElement>) => {

               <textarea

                 onInput={handleInput}
                 className={`w-full min-h-[88px] border-retro-thick ${error ? <threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user → RetroFileInput | File objects enter client process from the browser file picker (names, sizes, MIME types) |
| form values → RHF state | Values from Textarea/Checkbox flow into react-hook-form state, later submitted to backend API |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-57-01-01 | Tampering | RetroFileInput | mitigate | Enforce `accept="image/jpeg,image/png,image/heic"` and `maxSizeBytes` client-side; both are UX gates only — server remains authoritative per RESEARCH §Security Domain |
| T-57-01-02 | Denial of Service | RetroFileInput | mitigate | Reject files >10 MB before surfacing in `value` to prevent accidental OOM on preview rendering in later phases |
| T-57-01-03 | Injection (XSS) | RetroFileInput chip label | mitigate | Render `file.name` via React text child (auto-escaped); never `dangerouslySetInnerHTML` |
| T-57-01-04 | Info Disclosure | aria-label for remove button | accept | Filename is already visible in the chip; echoing into aria-label has no new disclosure |
| T-57-01-05 | Tampering | RetroCheckbox indeterminate | accept | Native HTMLInputElement state — no trust boundary crossed |
</threat_model>
