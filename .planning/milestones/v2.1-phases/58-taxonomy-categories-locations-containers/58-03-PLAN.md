    ```tsx
    import { useForm } from "react-hook-form";
    import { zodResolver } from "@hookform/resolvers/zod";
    import { useLingui } from "@lingui/react/macro";
    import { RetroFormField, RetroInput, RetroCombobox, RetroTextarea, type RetroOption } from "@/components/retro";
    import { categoryCreateSchema, type CategoryCreateValues } from "./schemas";

    export interface CategoryFormProps {
      defaultValues?: Partial<CategoryCreateValues>;
      parentOptions: RetroOption[]; // { value: id, label: name } filtered to exclude self+descendants
      onSubmit: (values: CategoryCreateValues) => void | Promise<void>;
      onDirtyChange?: (isDirty: boolean) => void;
      formId?: string; // for external submit button
    }

    export function CategoryForm({ defaultValues, parentOptions, onSubmit, onDirtyChange, formId }: CategoryFormProps) {
      const { t } = useLingui();
      const { control, handleSubmit, formState } = useForm<CategoryCreateValues>({
        resolver: zodResolver(categoryCreateSchema),
        defaultValues: { name: "", ...defaultValues },
        mode: "onBlur",
      });

      // Notify parent of dirty state for unsaved-changes guard
      useEffect(() => { onDirtyChange?.(formState.isDirty); }, [formState.isDirty, onDirtyChange]);

      const submit = handleSubmit((values) => {
        // strip empty optionals
        const cleaned: CategoryCreateValues = {
          name: values.name,
          parent_category_id: values.parent_category_id || undefined,
          description: values.description || undefined,
        };
        return onSubmit(cleaned);
      });

      return (
        <form id={formId} onSubmit={submit} className="flex flex-col gap-md">
          <RetroFormField name="name" control={control} label={t`NAME`}>
            <RetroInput autoFocus />
          </RetroFormField>
          <RetroFormField name="parent_category_id" control={control} label={t`PARENT CATEGORY`}>
            <RetroCombobox options={parentOptions} placeholder={t`Search categories…`} />
          </RetroFormField>
          <RetroFormField name="description" control={control} label={t`DESCRIPTION`}>
            <RetroTextarea rows={3} />
          </RetroFormField>
        </form>
      );
    }
    ```

    **File 3: `frontend2/src/features/taxonomy/forms/ContainerForm.tsx`**

    Same structure with `containerCreateSchema` + `ContainerCreateValues`. Fields: `name`, `short_code` (mono, auto-fill same as Location), `location_id` (required — `RetroCombobox` async searches locations), `description`. Labels: `NAME`, `SHORT CODE`, `LOCATION`, `DESCRIPTION`. Props include a `locationOptions: RetroOption[]` passed in.

    **CRITICAL — short_code submission strategy (RESEARCH Q2 RESOLVED):** The backend enforces per-workspace uniqueness on `short_code` for both Locations and Containers (`ErrShortCodeTaken`) and auto-generates an 8-char alphanumeric code when the field is empty. A client-generated 3-digit suffix has a >1% collision rate, which is unacceptable. Strategy:

    - `useAutoShortCode` returns `{ autoLinked: boolean }`. When `autoLinked === true`, the submit handler MUST strip `short_code` from the payload (set to `undefined`) so the backend auto-generates a guaranteed-unique code. Only when the user has manually typed into the short_code field (`autoLinked === false`) do we submit the user-provided value as-is.
    - In the LocationForm/ContainerForm submit handler, branch: `short_code: autoLinked ? undefined : values.short_code`.
    - The displayed auto-fill (e.g. "GAR-537") is therefore purely an editable preview — it is OVERWRITTEN by the backend-generated code on successful create unless the user edited it. Display the returned `short_code` in the success toast or panel confirmation so the user sees the final value (leave actual toast wording to Plan 02 — mutation already passes through `Location saved.`).

    **CRITICAL — parent clearing constraint (RESEARCH Q3 RESOLVED):** The backend's `UpdateCategoryInput.parent_category_id` and `UpdateLocationInput.parent_location` are typed `*uuid.UUID` with `omitempty`. Both absent and JSON `null` unmarshal to a nil pointer → "no change" semantics. There is NO way via the current API to move a node to root after it has a parent set. Plan 03 forms MUST therefore:

    - In the EDIT flow, the parent picker (`RetroCombobox`) MUST NOT be cleared once a parent was set — disable the combobox's clear-selection affordance when `defaultValues.parent_*` is set. If RetroCombobox has no built-in "disable clear" flag, wrap in a controller that rejects empty selections and shows a small helper `t\`Cannot move to root — contact support.\`` beneath the field when user attempts to clear.
    - In the submit handler, OMIT the `parent_*` key entirely from the PATCH payload if its value equals the original (or equals `undefined`). Only include it when the user selected a different (non-null) parent. Use RHF's `formState.dirtyFields.parent_*` as the signal.
    - Document this as a known limitation in `58-04-SUMMARY.md` for the UAT checkpoint.

    import { useLingui } from "@lingui/react/macro";
    import { RetroConfirmDialog, type RetroConfirmDialogHandle, RetroButton } from "@/components/retro";
    import { HttpError } from "@/lib/api";

    export type EntityKind = "category" | "location" | "container";

    export interface ArchiveDeleteFlowProps {
      entityKind: EntityKind;
      nodeName: string;
      onArchive: () => Promise<unknown>; // caller wires to useArchive*.mutateAsync
      onDelete: () => Promise<unknown>;  // caller wires to useDelete*.mutateAsync

      open: () => void;   // opens the archive dialog
      close: () => void;  // closes both
    }

    export const ArchiveDeleteFlow = forwardRef<ArchiveDeleteFlowHandle, ArchiveDeleteFlowProps>(
      function ArchiveDeleteFlow({ entityKind, nodeName, onArchive, onDelete }, ref) {
        const { t } = useLingui();
        const archiveRef = useRef<RetroConfirmDialogHandle>(null);
        const deleteRef = useRef<RetroConfirmDialogHandle>(null);

        useImperativeHandle(ref, () => ({
          open: () => archiveRef.current?.open(),
          close: () => {
            archiveRef.current?.close();
            deleteRef.current?.close();
          },
        }));

        // Labels per entity per 58-UI-SPEC.
        // RESEARCH Q4 RESOLVED: Lingui CLI cannot statically extract dynamic `t`${x}`` interpolation
        // into catalog keys. Use discriminated literals so each string is a standalone catalog entry.
        const archiveLabel =
          entityKind === "category" ? t`ARCHIVE CATEGORY`
          : entityKind === "location" ? t`ARCHIVE LOCATION`
          : t`ARCHIVE CONTAINER`;
        const deleteLabel =
          entityKind === "category" ? t`DELETE CATEGORY`
          : entityKind === "location" ? t`DELETE LOCATION`
          : t`DELETE CONTAINER`;

        const handleArchive = async () => {
          try {
            await onArchive();
            archiveRef.current?.close();
          } catch {
            // mutation hook already emitted a toast; keep dialog open so user sees retry context
          }
        };

        const handleDelete = async () => {
          try {
            await onDelete();
            deleteRef.current?.close();
          } catch (err) {
            // 409 short-circuit: close dialog; toast is handled by useDeleteCategory onError
            if (err instanceof HttpError && err.status === 409) {
              deleteRef.current?.close();
              archiveRef.current?.close();
              return;
            }
            // other errors: keep dialog open for retry
          }
        };

        const switchToDelete = () => {
          archiveRef.current?.close();
          // small delay to let native <dialog> transitions settle
          setTimeout(() => deleteRef.current?.open(), 0);
        };

        return (
          <>
            <RetroConfirmDialog
              ref={archiveRef}
              variant="soft"
              title={t`CONFIRM ARCHIVE`}
              body={t`This will hide '${nodeName}' from item pickers. You can restore it later.`}
              headerBadge={t`HIDES FROM PICKERS`}
              escapeLabel={t`← BACK`}
              destructiveLabel={archiveLabel}
              onConfirm={handleArchive}
              secondaryLink={{
                label: t`delete permanently`,
                onClick: switchToDelete,
              }}
            />
            <RetroConfirmDialog
              ref={deleteRef}
              variant="destructive"
              title={t`CONFIRM DELETE`}
              body={t`This permanently deletes '${nodeName}'. This action cannot be undone.`}
              escapeLabel={t`← BACK`}
              destructiveLabel={deleteLabel}
              onConfirm={handleDelete}
            />
          </>
        );
      },
    );
