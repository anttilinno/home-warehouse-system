    // Returns two maps keyed by inventory_id (item) and borrower_id respectively.
    // The ServiceInterface MUST provide access to item and borrower services; if it doesn't,

    // RegisterRoutes alongside svc. See Path B.
    func lookupLoanDecorations(
        ctx context.Context,
        lookup DecorationLookup, // see below
        workspaceID uuid.UUID,
        loans []*Loan,
    ) (map[uuid.UUID]LoanEmbeddedItem, map[uuid.UUID]LoanEmbeddedBorrower, error) {
        if len(loans) == 0 {
            return map[uuid.UUID]LoanEmbeddedItem{}, map[uuid.UUID]LoanEmbeddedBorrower{}, nil
        }
        itemIDs := make([]uuid.UUID, 0, len(loans))
        borrowerIDs := make([]uuid.UUID, 0, len(loans))
        seenItem := map[uuid.UUID]struct{}{}
        seenBor := map[uuid.UUID]struct{}{}
        for _, l := range loans {
            if _, ok := seenItem[l.InventoryID()]; !ok {
                itemIDs = append(itemIDs, l.InventoryID())
                seenItem[l.InventoryID()] = struct{}{}
            }
            if _, ok := seenBor[l.BorrowerID()]; !ok {
                borrowerIDs = append(borrowerIDs, l.BorrowerID())
                seenBor[l.BorrowerID()] = struct{}{}
            }
        }

        // Batch lookups â€” one SQL round-trip per concern
        itemsByID, err := lookup.ItemsByInventoryIDs(ctx, workspaceID, itemIDs)
        if err != nil {
            return nil, nil, err
        }
        primariesByItem, err := lookup.PrimaryPhotoThumbnailURLsByInventoryIDs(ctx, workspaceID, itemIDs)
        if err != nil {
            return nil, nil, err
        }
        borrowersByID, err := lookup.BorrowersByIDs(ctx, workspaceID, borrowerIDs)
        if err != nil {
            return nil, nil, err
        }

        itemMap := make(map[uuid.UUID]LoanEmbeddedItem, len(itemsByID))
        for id, name := range itemsByID {
            thumb := primariesByItem[id] // *string, nil-safe
            itemMap[id] = LoanEmbeddedItem{ID: id, Name: name, PrimaryPhotoThumbnailURL: thumb}
        }
        borrowerMap := make(map[uuid.UUID]LoanEmbeddedBorrower, len(borrowersByID))
        for id, name := range borrowersByID {
            borrowerMap[id] = LoanEmbeddedBorrower{ID: id, Name: name}
        }
        return itemMap, borrowerMap, nil
    }
