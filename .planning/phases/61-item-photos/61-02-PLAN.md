```typescript
import type { ItemPhoto } from "@/lib/api/itemPhotos";
import { HazardStripe } from "@/components/retro";
import { useLingui } from "@lingui/react/macro";

export interface ItemPhotoTileProps {
  photo: ItemPhoto;
  isPrimary: boolean;
  onClick: () => void;
}

export function ItemPhotoTile({ photo, isPrimary, onClick }: ItemPhotoTileProps) {
  const { t } = useLingui();
  const showPlaceholder =
    !photo.thumbnail_url ||
    photo.thumbnail_status === "pending" ||
    photo.thumbnail_status === "processing";

  return (
    <button
      type="button"
      onClick={onClick}
      aria-label={t`Open photo ${photo.filename}`}
      className="relative block w-full aspect-square border-retro-thick border-retro-charcoal bg-retro-cream cursor-pointer overflow-hidden focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-retro-amber hover:shadow-retro-raised"
    >
      {showPlaceholder ? (
        <div className="relative w-full h-full">
          <HazardStripe className="absolute inset-0 h-full opacity-60" height={undefined} />
          <span className="absolute inset-0 flex items-center justify-center font-mono text-[12px] uppercase text-retro-charcoal font-bold">
            {t`PROCESSING…`}
          </span>
        </div>
      ) : (
        <img
          src={photo.thumbnail_url}
          alt=""
          loading="lazy"
          className="w-full h-full object-cover"
        />
      )}
      {isPrimary && (
        <span className="absolute top-xs left-xs px-xs py-[2px] bg-retro-amber text-retro-cream font-sans text-[12px] font-bold uppercase border border-retro-charcoal">
          ★ {t`PRIMARY`}
        </span>
      )}
    </button>
  );
}
```

Key points per 61-PATTERNS.md + 61-RESEARCH.md:
- Use a `<button>` not a `<div>` for the tile root: keyboard focus trap, native click semantics, Tab reachability.
- `HazardStripe` gets `height={undefined}` + `className="h-full"` to fill the tile (Pitfall 5 fix).
- Primary badge uses the 12px-typography exception (documented in UI-SPEC).
- All visible strings pass through `t` macro.

```typescript
import type { ItemPhoto } from "@/lib/api/itemPhotos";
import { ItemPhotoTile } from "./ItemPhotoTile";

export interface ItemPhotoGridProps {
  photos: ItemPhoto[];
  onTileClick: (index: number) => void;
}

export function ItemPhotoGrid({ photos, onTileClick }: ItemPhotoGridProps) {
  return (
    <div
      className="grid grid-cols-3 gap-sm md:gap-md lg:grid-cols-4"
      role="list"
    >
      {photos.map((photo, index) => (
        <div role="listitem" key={photo.id}>
          <ItemPhotoTile
            photo={photo}
            isPrimary={photo.is_primary}
            onClick={() => onTileClick(index)}
          />
        </div>
      ))}
    </div>
  );
}
```

- Grid does NOT own empty-state rendering — the parent `ItemPhotoGallery` (Plan 61-03) makes that call.
