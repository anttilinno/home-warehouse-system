**Save flow (handleSave):**

```typescript
async function handleSave() {
  if (!name.trim() || photos.length === 0 || isSaving) return;
  setIsSaving(true);

  try {
    const sku = generateSKU();
    const tempId = await mutate({
      sku,
      name: name.trim(),
      needs_review: true,
      category_id: settings.categoryId || undefined,
      location_id: settings.locationId || undefined,
    } as Record<string, unknown>);

    // Store photos in IndexedDB linked to tempId
    for (const photo of photos) {
      await storePhoto(tempId, photo.blob);
    }

    // Feedback (QC-08)
    triggerHaptic("success");
    playSuccessBeep();
    incrementCaptureCount();

    // Reset form for next item (QC-06)
    photos.forEach(p => URL.revokeObjectURL(p.preview));
    setPhotos([]);
    setName("");

    // Re-trigger camera for next capture
    setTimeout(() => cameraInputRef.current?.click(), 100);
  } finally {
    setIsSaving(false);
  }
}
```
