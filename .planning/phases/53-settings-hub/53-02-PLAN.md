```typescript
const handleAvatarUpload = async (e: React.ChangeEvent<HTMLInputElement>) => {
  const file = e.target.files?.[0];
  if (!file) return;
  const formData = new FormData();
  formData.append("avatar", file);
  try {
    await fetch("/api/users/me/avatar", {
      method: "POST",
      credentials: "include",
      body: formData,
      // DO NOT set Content-Type header — browser sets it with boundary
    });
    await refreshUser();
    addToast(t`AVATAR UPLOADED`, "success");
  } catch {
    addToast(t`Failed to upload avatar`, "error");
  }
  // Reset file input so same file can be re-selected
  e.target.value = "";
};
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client->API (profile) | Name, email, avatar file uploaded to backend |
| client->API (password) | Current + new password sent to backend |
| client->API (sessions) | Session revocation requests |
| client->API (OAuth) | OAuth link/unlink flow |
| client->API (deletion) | Irreversible account deletion |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-53-03 | Tampering | ProfilePage avatar upload | mitigate | Use raw fetch with credentials:include (HttpOnly cookie auth); backend validates file type and size |
| T-53-04 | Elevation of Privilege | SecurityPage OAuth unlink | mitigate | Disable UNLINK when `accounts.length === 1 && !user.has_password` to prevent account lockout |
| T-53-05 | Denial of Service | SecurityPage session revoke | mitigate | Hide REVOKE button for `is_current === true` sessions; backend returns 400 as backup |
| T-53-06 | Tampering | SecurityPage account deletion | mitigate | RetroDialog confirmation required before DELETE /users/me; action is intentional |
| T-53-07 | Information Disclosure | SecurityPage sessions list | accept | Sessions are only the current user's own sessions; backend scopes by JWT identity |
| T-53-08 | Spoofing | SecurityPage password change | mitigate | Backend requires `current_password` verification before accepting new password |
</threat_model>
